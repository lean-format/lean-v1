Quick Start

Get started with LEAN in minutes

1. Installation

npm install lean-format

2. Node.js Usage

const { parse, format } = require('lean-format');

// Parse LEAN to JavaScript
const  data = parse(`
users(id, name, age):
    - 1, Alice, 30
    - 2, Bob, 25
`);

// Format JavaScript as LEAN
const  lean = format(data);
console.log(lean);
